#! /usr/bin/env python3

from ncclient import manager

# Router credentials
ROUTER = "192.0.2.1"
PORT = 830
USERNAME = "netconf-user"
PASSWORD = "netconf_pass"


# NETCONF configuration (in XML)
config = """
<configuration>
  <routing-options>
    <static>
      <route>
        <name>203.0.113.0/24</name>
        <next-hop>192.0.2.254</next-hop>
      </route>
    </static>
  </routing-options>
</configuration>
"""    

def add_static_route():
    with manager.connect(
        host=ROUTER,
        port=PORT,
        username=USERNAME,
        password=PASSWORD,
        hostkey_verify=False
    ) as m:

        # Push configuration to router
        m.edit_config(target='candidate', config=config)
        m.commit()


if __name__ == "__main__":
    add_static_route()
    print("Static route configured.")
